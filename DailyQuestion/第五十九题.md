#### 第 59 题：模拟new实现

```javascript
        function Foo(name, age) {
          this.name = name;
          this.age = age
        }

        var nar = new Foo('tom', 18)
        console.log(nar.name)
        function OBK() {
          var obj = new Object();
          newArgument = [].shift.call(arguments);
          obj.__proto__ = newArgument.prototype;
          var ret  = newArgument.apply(obj, arguments)

          return typeof ret === 'object' ? ret : obj;
        }
        var bar = OBK(Foo, 'jim', 15)
        console.log(bar.age)
```
