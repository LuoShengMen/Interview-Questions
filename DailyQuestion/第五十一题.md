考察隐式转换,重写toString方法即可
```javascript
  var a = {
    i: 1,
    toString() {
      return a.i++;
    }
  }

  if(a == 1 && a == 2 && a == 3){
    console.log(1);
  }
```

```javascript
let a = {
  i: 1,
  valueOf () {
    return a.i++
  }
}

if(a == 1 && a == 2 && a == 3) {
  console.log('1');
}
```

```javascript
var a = [1,2,3];
a.join = a.shift;
if(a == 1 && a == 2 && a == 3) {
  console.log('1');
}
```

let a = {[Symbol.toPrimitive]: ((i) => () => ++i) (0)};

  console.log('1');
}
