#### 第 34 题：简单改造下面的代码，使之分别打印 10 和 20

```
var b = 10;
(function b(){
    b = 20;
    console.log(b); 
})();
```
* 打印20
1.
```
  var b = 10;
  (function b(){
    var b = 20;
    console.log(b);
  })();// 20 在自执行函数中重新定义一个变量，
```
2.
```
var b = 10;
(function a(){
   b = 20;
   console.log(b); 
})();
```

3.
```
var b = 10;
(function (){
   b = 20;
   console.log(b); 
})();
```


* 打印10
1.
```
  var b = 10;
  (function b(){
    b = 20;
    console.log(window.b);
  })();
  
```
// 在自执行函数中访问window，window中的b值为10
2.
```
var b = 10;
(function (){
  console.log(b);
  b = 20;
})();
```

3.
```
var b = 10;
(function b(b){
   console.log(b); 
   b = 20;
})(b);
```
